#!/bin/sh

# Define the rule criteria and the new rule to be added
OLD_RULE="-p tcp --dport 8080 -j ACCEPT"
NEW_RULE="-p tcp --dport 9090 -j ACCEPT"

# Delete the old rule based on its specification
echo "Deleting the old iptables rule..."
sudo iptables -D INPUT $OLD_RULE

# Add the new rule
echo "Adding the new iptables rule..."
sudo iptables -A INPUT $NEW_RULE

# Verify the changes
echo "Current iptables rules:"
sudo iptables -L -n -v

echo "iptables rule modified successfully."
